f=open('h.txt','r')
w=[]
for i in f:
    temp_list=i.split()
    for j in temp_list:
        w.append(j)
    temp_list.clear()

custom_word=input('Enter a word to be searched in file: ')
count_for_word=0
for z in w:
    if z == custom_word:
        count_for_word=count_for_word+1

print('The word '+custom_word+' was found '+str(count_for_word)+' times in the specified file')
